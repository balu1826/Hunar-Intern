import pandas as pd
df=pd.read_csv('food_coded.csv')
print(df.head())
#initial shape of dataframe
print(df.shape)
#seperating numeric columns
numeric_cols=df.select_dtypes(include=['number']).columns
#seperating non-numeric columns
non_numeric_cols=df.select_dtypes(exclude=['number']).columns
#applying mean and median for numeric values
df[numeric_cols]=df[numeric_cols].fillna(df[numeric_cols].mean())
df[numeric_cols]=df[numeric_cols].fillna(df[numeric_cols].median())
#applying mode for non numeric values
for col in non_numeric_cols:
    df[col] = df[col].fillna(df[col].mode().iloc[0])

#removing duplicate values
no_duplicates=df.drop_duplicates()
#converting rows into columns to remove duplicate ones
transposed=no_duplicates.T
#removing duplicate rows
no_transposed_duplicates=transposed.drop_duplicates()
#back to norm form
back_to_columns=no_transposed_duplicates.T
print("After cleaning and removing the data")
print(df.head())
print(df.shape)

print("Missing values after cleaning")
print(back_to_columns.isnull().sum())
